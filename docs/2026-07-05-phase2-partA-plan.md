```python
KEEP_EMPTY = {"meetings", "inbox"}  # intentional dirs, never flagged as empty scaffold

def _empty_scaffold_dirs(project: Path) -> list[str]:
    out = []
    for d in project.rglob("*"):
        if ".git" in d.parts or not d.is_dir():
            continue
        if d.name in KEEP_EMPTY:
            continue
        entries = [e for e in d.iterdir() if e.name != ".DS_Store"]
        only_readme = entries and all(e.name == "README.md" for e in entries)
        if not entries or only_readme:
            out.append(f"empty scaffold dir: {d.relative_to(project).as_posix()}")
    return out

def _undrained(project: Path) -> list[str]:
    out = []
    inbox = project / "memory" / "inbox"
    if inbox.is_dir():
        for f in sorted(inbox.glob("*.md")):
            if f.name == "README.md":
                continue
            text = f.read_text()
            status = next((ln.split(":", 1)[1].strip().lower()
                           for ln in text.splitlines()
                           if ln.lower().lstrip("*# ").startswith("status:")), "")
            if status != "drained":
                out.append(f"un-drained inbox: {f.relative_to(project).as_posix()}")
    cap = project / "capture.md"
    if cap.is_file():
        body = "\n".join(ln for ln in cap.read_text().splitlines()
                         if ln.strip() and not ln.startswith("#")
                         and ln.strip() not in ("- ", "-"))
        if "braindump" not in body.lower() and len(body) > 40:
            out.append("un-drained capture: capture.md")
    return out
```

```python
def parse_topic_meta(path: Path) -> dict:
    text = path.read_text()
    summary, tags, title = "", [], path.stem
    lines = text.splitlines()
    body_start = 0
    if lines and lines[0].strip() == "---":
        for i in range(1, len(lines)):
            if lines[i].strip() == "---":
                body_start = i + 1
                break
            key, _, val = lines[i].partition(":")
            key, val = key.strip().lower(), val.strip()
            if key == "summary":
                summary = val
            elif key == "tags":
                inner = val.strip().lstrip("[").rstrip("]")
                tags = [t.strip() for t in inner.split(",") if t.strip()]
    for ln in lines[body_start:]:
        s = ln.strip()
        if s.startswith("# "):
            title = s[2:].strip()
            break
    if not summary:
        for ln in lines[body_start:]:
            s = ln.strip().lstrip("-* ").strip()
            if s and not s.startswith("#"):
                summary = s
                break
    return {"title": title, "summary": summary, "tags": tags, "path": path}
```

```python
def build_repo_catalog(root: Path) -> str:
    lines = [GENERATED_HEADER, "", "# Projects Catalog", ""]
    for project in discover_projects(root):
        lines.append(f"## {project.name}")
        lines.append("")
        metas = _topic_metas(project)
```

```python
#!/usr/bin/env python3
"""Keyword lookup over curated agent-memory: returns ranked file:line pointers."""
from __future__ import annotations
import argparse
import re
from pathlib import Path
from memory_lib import resolve_root, discover_projects, parse_topic_meta

def _terms(query: str) -> list[str]:
    return [t for t in re.findall(r"[a-z0-9]+", query.lower()) if len(t) > 1]

def _best_line(text: str, terms: list[str]) -> tuple[int, str, int]:
    best_line, best_snip, best_hits = 0, "", 0
    for i, ln in enumerate(text.splitlines(), 1):
        low = ln.lower()
        hits = sum(low.count(t) for t in terms)
        if hits > best_hits:
            best_line, best_snip, best_hits = i, ln.strip()[:120], hits
    return best_line, best_snip, best_hits

def ask(root: Path, query: str, project: str | None = None, limit: int = 8) -> list[dict]:
    terms = _terms(query)
    if not terms:
        return []
    projects = discover_projects(root)
    if project:
        projects = [p for p in projects if p.name == project]
    results = []
    for proj in projects:
        tdir = proj / "memory" / "topics"
        if not tdir.is_dir():
            continue
        for tp in sorted(tdir.glob("*.md")):
            if tp.name == "README.md":
                continue
            meta = parse_topic_meta(tp)
            text = tp.read_text()
            line, snip, body_hits = _best_line(text, terms)
            meta_blob = (meta["title"] + " " + meta["summary"] + " " + " ".join(meta["tags"])).lower()
            meta_hits = sum(meta_blob.count(t) for t in terms)
            score = body_hits + 3 * meta_hits
            if score:
                results.append({
                    "path": str(tp.relative_to(root)),
                    "line": line or 1,
                    "score": score,
                    "snippet": snip or meta["summary"],
                })
    results.sort(key=lambda r: (-r["score"], r["path"]))
    return results[:limit]

def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("query")
    ap.add_argument("--project")
    ap.add_argument("--root")
    args = ap.parse_args()
    root = resolve_root(args.root)
    hits = ask(root, args.query, args.project)
    if not hits:
        print("no matches")
        return 0
    for h in hits:
        print(f"{h['path']}:{h['line']}  (score {h['score']})  {h['snippet']}")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
```
